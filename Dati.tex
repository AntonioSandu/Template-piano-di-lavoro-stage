\newcommand{\prospettoSettimanale}{
     % Personalizzare indicando in lista, i vari task settimana per settimana
     % sostituire a XX il totale ore della settimana
    \begin{itemize}
        \item \textbf{Prima Settimana (32 ore)}
        \begin{itemize}
            \item Incontro con persone coinvolte nel progetto per discutere i requisiti e le richieste
            relativamente al sistema da sviluppare;
            \item Formazione autonoma sui framework front-end moderni;
        \end{itemize}
        \item \textbf{Seconda Settimana - Formazione (40 ore)} 
        \begin{itemize}
            \item Continuazione e conclusione della formazione sui framework front-end;
            \item Inizio attività di analisi dei requisiti;
            \item Allineamento con il tutor aziendale sui risultati raggiunti fino a questo punto;
        \end{itemize}
        \item \textbf{Terza Settimana - Analisi dei Requisiti (40 ore)} 
        \begin{itemize}
            \item Continuazione attività di analisi dei requisiti;
            \item Inizio stesura documentazione relativa ad analisi dei requisiti;
        \end{itemize}
        \item \textbf{Quarta Settimana - Progettazione (40 ore)} 
        \begin{itemize}
            \item Completamento stesura del documento di Analisi dei Requisiti;
            \item Inizio attività di progettazione;
            \item Validazione con il tutor aziendale dei risultati raggiunti fino a questo punto;
        \end{itemize}
        \item \textbf{Quinta Settimana - Progettazione nel dettaglio (40 ore)} 
        \begin{itemize}
            \item Continuazione attività di progettazione;
            \item Inizio stesura documentazione relativa alla progettazione;
            \item Validazione con il tutor aziendale dei risultati raggiunti fino a questo punto;
        \end{itemize}
        \item \textbf{Sesta Settimana - Codifica (40 ore)} 
        \begin{itemize}
            \item Inizio attività di codifica del back-end della piattaforma;
            \item Test di unità per il back-end della piattaforma;
            \item Inizio attività di codifica del front-end della piattaforma;
            \item Test di unità per il front-end della piattaforma;
        \end{itemize}
        \item \textbf{Settima Settimana - Conclusione della codifica (40 ore)} 
        \begin{itemize}
            \item Conclusione attività di codifica del back-end della piattaforma;
            \item Conclusione attività di codifica del front-end della piattaforma;
            \item Conclusione test di unità per il back-end e front-end della piattaforma;
            \item Inizio attività di collaudo della piattaforma;
        \end{itemize}
        \item \textbf{Ottava Settimana - Conclusione (40 ore)} 
        \begin{itemize}
            \item Completamento stesura del documento di Specifica Tecnica;
            \item Ultimo incontro di validazione degli artefatti con il tutor aziendale;
        \end{itemize}
    \end{itemize}
}
